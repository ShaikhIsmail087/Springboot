    /**
        Method Name 					: executeUpdate()
        Arguments						: String sqlQuery, Object[] bindParams
        Returns						    : boolean updateFlag
        Comments						: Returns an ArrayList after executing the sql statement
    */

    public boolean executeUpdate(String sqlQuery, Object[] bindParams) 
        throws Exception {


        PreparedStatement pst = null;
        boolean  updateFlag = false;
        Connection conn = null;

        int updateCount = 0;

        try {

            conn = getConnection();
            pst = conn.prepareStatement(sqlQuery);

			Log.printLog(application, "SQLUtility.java: executeUpdate(): query : "+sqlQuery, 1);

            if (bindParams != null) {
            
                for (int i = 0; i < bindParams.length; i++) {

                    if(bindParams[i] == null) {

                      pst.setNull(i+1,Types.JAVA_OBJECT);
                    }else{

                      pst.setObject(i+1,bindParams[i]);
                    }
                }
            }


            updateCount   = pst.executeUpdate();

			Log.printLog(application, "SQLUtility.java: executeUpdate(): updateCount : "+updateCount, 1);

            if (updateCount > 0) {

                updateFlag = true;
            }

        } finally {

            closeConnection(null, pst, conn);
		}

        return updateFlag;
    }
